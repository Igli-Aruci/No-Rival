var output = {
    rewards: []
};
var sku = "COP";

// Accediamo direttamente all'account
var account = ikContext.account;

if (account && account.isDraft === true) {
    var targetPax = account.clientCount || 0;
    
    // Conteggio manuale delle righe esistenti con quello SKU
    var existingCopCount = 0;
    var lines = account.transactionLines || [];

    for (var i = 0; i < lines.length; i++) {
        // Verifichiamo lo SKU in tutti i modi possibili in cui Lightspeed lo chiama
        if (lines[i].itemSku === sku || lines[i].sku === sku || lines[i].productSku === sku) {
            existingCopCount++;
        }
    }

    // Se mancano righe rispetto ai pax impostati
    if (targetPax > existingCopCount) {
        var diff = targetPax - existingCopCount;
        
        for (var j = 0; j < diff; j++) {
            output.rewards.push({
                type: "ADD",
                sku: sku,
                // Proviamo a non specificare la fase per lasciare che il POS decida
                // oppure usa 4 se il sistema lo richiede strettamente
                insertionPhase: 4 
            });
        }
    }
}

// Restituiamo il JSON finale
JSON.stringify(output);
