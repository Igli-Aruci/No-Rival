var output = {
    rewards: []
};
var sku = "COP";

// 1. FORZATURA DEL TRIGGER: leggiamo il totale e il numero pax 
// per assicurarci che lo script veda ogni minima variazione.
var currentPax = ikContext.account.clientCount || 0;
var totalCheck = ikContext.account.totalAmount || 0; 

if (ikContext.account.isDraft === true) {
    
    // 2. CONTEGGIO PUNTUALE
    var lines = ikContext.account.transactionLines || [];
    var copertiPresenti = 0;

    for (var i = 0; i < lines.length; i++) {
        if (lines[i].itemSku === sku) {
            copertiPresenti++;
        }
    }

    // 3. LOGICA DI AGGIORNAMENTO (SOLO AUMENTO)
    // Se il numero attuale nel conto è minore di quello dei pax, aggiungi.
    if (currentPax > copertiPresenti) {
        var diff = currentPax - copertiPresenti;
        
        for (var a = 0; a < diff; a++) {
            output.rewards.push({
                type: "ADD",
                sku: sku,
                // Usiamo la fase 4 per farlo inserire dopo le righe normali
                insertionPhase: 4 
            });
        }
    }
}

// 4. ESECUZIONE
JSON.stringify(output);
