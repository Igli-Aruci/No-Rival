var output = {
    rewards: []
};
var sku = "COP";

// Verifichiamo che l'account sia attivo (Draft)
if (ikContext.account && ikContext.account.isDraft === true) {
    
    // 1. Quanti pax (coperti) sono impostati ora sul tavolo?
    var targetPax = ikContext.account.clientCount || 0;
    
    // 2. Controlliamo cosa c'è già nel conto (transactionLines)
    var lines = ikContext.account.transactionLines || [];
    var copertiGiaPresenti = [];

    for (var i = 0; i < lines.length; i++) {
        if (lines[i].itemSku === sku) {
            copertiGiaPresenti.push(lines[i]);
        }
    }

    var countAttuale = copertiGiaPresenti.length;

    // 3. LOGICA DI AGGIORNAMENTO
    if (targetPax > countAttuale) {
        // Se i pax aumentano, AGGIUNGIAMO la differenza
        var daAggiungere = targetPax - countAttuale;
        for (var a = 0; a < daAggiungere; a++) {
            output.rewards.push({
                type: "ADD",
                sku: sku,
                insertionPhase: 4 // Fase di automazione per Lightspeed
            });
        }
    } 
    else if (targetPax < countAttuale) {
        // Se i pax diminuiscono, RIMUOVIAMO l'eccedenza
        var daRimuovere = countAttuale - targetPax;
        for (var r = 0; r < daRimuovere; r++) {
            // Per rimuovere, Lightspeed ha bisogno dell'ID unico della riga
            output.rewards.push({
                type: "REMOVE",
                id: copertiGiaPresenti[r].id
            });
        }
    }
}

// Restituisce il comando finale
JSON.stringify(output);
