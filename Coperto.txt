var output = { rewards: [] };
var sku = "COP";

if (ikContext.account && ikContext.account.isDraft === true) {
    var targetCount = ikContext.account.clientCount || 0;
    var existingRewards = ikContext.account.rewards || [];
    
    // Filtriamo i premi esistenti per trovare tutti i "COP"
    var copItems = existingRewards.filter(function(item) {
        return item.sku === sku;
    });

    var currentCount = copItems.length;

    if (targetCount > currentCount) {
        // AGGIUNGI: Lightspeed di solito accetta ADD
        var toAdd = targetCount - currentCount;
        for (var i = 0; i < toAdd; i++) {
            output.rewards.push({
                type: "ADD",
                sku: sku
            });
        }
    } 
    else if (targetCount < currentCount) {
        // RIMOZIONE LIGHTSPEED: 
        // In molti moduli Lightspeed, il comando corretto è "VOID" o "DELETE"
        // e richiede il 'uuid' o il 'line_item_id'
        var toRemove = currentCount - targetCount;
        for (var j = 0; j < toRemove; j++) {
            output.rewards.push({
                type: "VOID", // Prova VOID, è lo standard Lightspeed per annullare righe
                id: copItems[j].id || copItems[j].uuid
            });
        }
    }
}

JSON.stringify(output);
