var output = {
    rewards: []
};
var sku = "COP";

// Verifichiamo che l'account sia in stato Bozza (Draft)
if (ikContext.account && ikContext.account.isDraft === true) {
    
    // 1. Quanti pax (coperti) sono segnati ora sul tavolo?
    var targetPax = ikContext.account.clientCount || 0;
    
    // 2. Contiamo quanti SKU "COP" sono già presenti nel carrello
    var lines = ikContext.account.transactionLines || [];
    var countAttuale = 0;

    for (var i = 0; i < lines.length; i++) {
        // Usiamo itemSku o sku a seconda di come lo passa Lightspeed
        if (lines[i].itemSku === sku || lines[i].sku === sku) {
            countAttuale++;
        }
    }

    // 3. LOGICA DI AUMENTO
    // Se i pax segnati sono più degli SKU presenti, aggiungiamo la differenza
    if (targetPax > countAttuale) {
        var daAggiungere = targetPax - countAttuale;
        
        for (var a = 0; a < daAggiungere; a++) {
            output.rewards.push({
                type: "ADD",
                sku: sku,
                insertionPhase: 4 // Fondamentale per Lightspeed
            });
        }
    }
}

// Restituisce il comando al POS
JSON.stringify(output);
