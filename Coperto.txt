var output = { rewards: [] };
var sku = "COP";

if (ikContext.account && ikContext.account.isDraft === true) {
    var targetCount = ikContext.account.clientCount || 0;

    // Inviamo un comando di tipo UPDATE o SET
    // Questo dice al sistema di ignorare i calcoli e impostare il totale
    output.rewards.push({
        type: "UPDATE", 
        sku: sku,
        quantity: targetCount
    });
}

JSON.stringify(output);
