var output = { rewards: [] };
var sku = "COP";

// 1. Verifichiamo la presenza dell'account e dei dati
if (ikContext && ikContext.account) {
    var account = ikContext.account;
    
    // Solo se siamo in bozza (tavolo aperto non pagato)
    if (account.isDraft === true) {
        
        var targetPax = account.clientCount || 0;
        
        // 2. SCANSIONE MANUALE: non usiamo filter(), cerchiamo riga per riga
        var currentCopCount = 0;
        if (account.transactionLines && account.transactionLines.length > 0) {
            for (var i = 0; i < account.transactionLines.length; i++) {
                var line = account.transactionLines[i];
                // Controlliamo sia itemSku che sku (per sicurezza tra versioni diverse)
                if (line.itemSku === sku || line.sku === sku) {
                    currentCopCount++;
                }
            }
        }

        // 3. LOGICA DI FORZATURA
        // Se i pax sul tavolo sono 4 e noi ne contiamo solo 2...
        if (targetPax > currentCopCount) {
            var diff = targetPax - currentCopCount;
            
            for (var j = 0; j < diff; j++) {
                output.rewards.push({
                    type: "ADD",
                    sku: sku,
                    // Usiamo una fase diversa (0 o 1) per forzare l'inserimento immediato
                    insertionPhase: 1 
                });
            }
        }
    }
}

// 4. RESTITUIAMO IL JSON
JSON.stringify(output);
